const root = document.body;
const editable = root.dataset.sharedEditorMode === "edit";
const editor = document.querySelector("[data-shared-source]");
const highlight = document.querySelector("[data-shared-highlight]");
const collaboratorSelections = document.querySelector("[data-shared-collaborator-selections]");
const saveStatus = editable ? document.querySelector("#edit-save-status") : document.querySelector("#shared-save-status");
const liveStatus = editable ? document.querySelector("#edit-live-status") : document.querySelector("#shared-live-status");
const pdf = document.querySelector("[data-shared-pdf-viewer]");
const surfaces = document.querySelector("#shared-editor-surfaces");
const layoutControl = document.querySelector("#shared-editor-layout");
const showSource = document.querySelector("#show-shared-source");
const showPdf = document.querySelector("#show-shared-pdf");
const savePath = root.dataset.sharedSavePath;
const snapshotPath = root.dataset.sharedSnapshotPath;
const socketPath = root.dataset.sharedSocketPath;
const fileId = root.dataset.sharedFileId;
let revision = Number(root.dataset.sharedRevision ?? "0");
let savedContent = editor?.value ?? "";
let saveTimer;
let saving = false;
let pending = false;
let blocked = false;
let socket;
let socketSynced = false;
let reconnectDelay = 500;
let reconnectTimer;
let refreshTimer;
let selectionBroadcastTimer;
const remoteSelections = new Map();

const setSaveStatus = (value) => {
  if (saveStatus) saveStatus.textContent = value;
};

const setLiveStatus = (value) => {
  if (liveStatus) liveStatus.textContent = value;
};

const setActiveSurface = (surface) => {
  if (surfaces) surfaces.dataset.activeSurface = surface;
  showSource?.setAttribute("aria-pressed", String(surface === "authoring"));
  showPdf?.setAttribute("aria-pressed", String(surface === "context"));
};

const setLayout = (value) => {
  const layout = value === "editor" || value === "pdf" ? value : "split";
  if (surfaces) surfaces.dataset.layout = layout;
  if (layoutControl) layoutControl.value = layout;
  if (layout === "pdf") setActiveSurface("context");
  else if (layout === "editor") setActiveSurface("authoring");
  window.dispatchEvent(new Event("resize"));
};

layoutControl?.addEventListener("change", () => setLayout(layoutControl.value));
showSource?.addEventListener("click", () => setActiveSurface("authoring"));
showPdf?.addEventListener("click", () => setActiveSurface("context"));
setLayout(layoutControl?.value ?? surfaces?.dataset.layout ?? "split");

const collaboratorColor = (collaboratorId) => {
  let hash = 0;
  for (const character of collaboratorId) hash = (hash * 31 + character.codePointAt(0)) >>> 0;
  return hash % 4;
};

const renderSource = () => {
  if (!editor || !highlight) return;
  const ranges = editable
    ? [...remoteSelections.values()]
        .filter((selection) => selection.fileId === fileId && selection.revision === revision && selection.end <= editor.value.length)
        .map((selection) => ({ ...selection, color: collaboratorColor(selection.collaboratorId) }))
    : [];
  const fragment = document.createDocumentFragment();
  let lineNumber = 1;
  let lineStart = 0;
  const appendLine = (lineEnd) => {
    const line = document.createElement("span");
    line.className = "source-editor-line";
    line.dataset.lineNumber = String(lineNumber);
    const boundaries = new Set([lineStart, lineEnd]);
    for (const range of ranges) {
      if (range.start >= lineStart && range.start <= lineEnd) boundaries.add(range.start);
      if (range.end >= lineStart && range.end <= lineEnd) boundaries.add(range.end);
    }
    const ordered = [...boundaries].sort((left, right) => left - right);
    for (let index = 0; index < ordered.length; index += 1) {
      const start = ordered[index];
      for (const range of ranges.filter((candidate) => candidate.start === start && candidate.end === start)) {
        const caret = document.createElement("span");
        caret.className = "collaborator-caret";
        caret.dataset.collaboratorColor = String(range.color);
        line.append(caret);
      }
      const end = ordered[index + 1];
      if (end === undefined || end === start) continue;
      const text = editor.value.slice(start, end);
      const selected = ranges.find((range) => range.start < end && range.end > start);
      if (!selected) {
        line.append(document.createTextNode(text));
        continue;
      }
      const mark = document.createElement("span");
      mark.className = "collaborator-selection";
      mark.dataset.collaboratorColor = String(selected.color);
      mark.textContent = text;
      line.append(mark);
    }
    fragment.append(line);
    lineNumber += 1;
  };
  for (const match of editor.value.matchAll(/\r\n|\r|\n/gu)) {
    appendLine(match.index);
    lineStart = match.index + match[0].length;
  }
  appendLine(editor.value.length);
  highlight.replaceChildren(fragment);
  if (collaboratorSelections) {
    collaboratorSelections.textContent = ranges
      .map((selection) =>
        selection.start === selection.end
          ? `Collaborator caret at ${selection.start}.`
          : `Collaborator selection from ${selection.start} to ${selection.end}.`,
      )
      .join(" ");
  }
};

const syncHighlightScroll = () => {
  if (!editor || !highlight) return;
  highlight.scrollTop = editor.scrollTop;
  highlight.scrollLeft = editor.scrollLeft;
};

const scheduleSelectionBroadcast = () => {
  if (!editable) return;
  window.clearTimeout(selectionBroadcastTimer);
  selectionBroadcastTimer = window.setTimeout(() => {
    if (!editor || !fileId || !socketSynced || socket?.readyState !== WebSocket.OPEN) return;
    socket.send(
      JSON.stringify({
        type: "selection",
        protocol: 1,
        fileId,
        start: editor.selectionStart,
        end: editor.selectionEnd,
        revision,
      }),
    );
  }, 80);
};

const isRemoteSelection = (value) =>
  value &&
  value.type === "selection" &&
  typeof value.collaboratorId === "string" &&
  value.collaboratorId.length > 0 &&
  value.collaboratorId.length <= 128 &&
  typeof value.fileId === "string" &&
  Number.isSafeInteger(value.start) &&
  Number.isSafeInteger(value.end) &&
  value.start >= 0 &&
  value.end >= value.start &&
  Number.isSafeInteger(value.revision) &&
  value.revision >= 0;

const endAccess = () => {
  blocked = true;
  if (editable && editor) editor.disabled = true;
  setLiveStatus(editable ? "Edit access ended" : "Live access ended");
  setSaveStatus(editable ? "Edit access ended" : `Read only · revision ${revision}`);
};

const scheduleRefresh = (nextRevision) => {
  revision = nextRevision;
  root.dataset.sharedRevision = String(revision);
  setLiveStatus(`Changes received · updating to revision ${revision}`);
  setSaveStatus(editable ? `Updating · revision ${revision}` : `Read only · revision ${revision}`);
  window.clearTimeout(refreshTimer);
  refreshTimer = window.setTimeout(() => location.reload(), 350);
};

const connect = () => {
  if (!socketPath || blocked) return;
  const protocol = location.protocol === "https:" ? "wss:" : "ws:";
  socket = new WebSocket(`${protocol}//${location.host}${socketPath}`);
  socket.addEventListener("open", () => {
    reconnectDelay = 500;
    setLiveStatus(editable ? "Joining live project…" : `Live · revision ${revision}`);
  });
  socket.addEventListener("message", (event) => {
    if (typeof event.data !== "string") return;
    let message;
    try {
      message = JSON.parse(event.data);
    } catch {
      return;
    }
    if (editable && message.type === "sync" && message.protocol === 1 && Number.isSafeInteger(message.revision)) {
      socketSynced = true;
      setLiveStatus("Live");
      if (message.revision > revision && !saving && editor?.value === savedContent) location.reload();
      else scheduleSelectionBroadcast();
      return;
    }
    if (editable && message.type === "presence" && Number.isSafeInteger(message.collaborators)) {
      setLiveStatus(`Live · ${message.collaborators} ${message.collaborators === 1 ? "writer" : "writers"}`);
      return;
    }
    if (editable && isRemoteSelection(message)) {
      if (message.revision === revision) remoteSelections.set(message.collaboratorId, message);
      renderSource();
      return;
    }
    if (editable && message.type === "selection-clear" && typeof message.collaboratorId === "string") {
      remoteSelections.delete(message.collaboratorId);
      renderSource();
      return;
    }
    if ((message.type !== "revision" && message.type !== "reset") || !Number.isSafeInteger(message.revision)) return;
    if (message.revision <= revision) return;
    remoteSelections.clear();
    renderSource();
    if (!editable || (!saving && editor?.value === savedContent)) scheduleRefresh(message.revision);
  });
  socket.addEventListener("close", (event) => {
    socketSynced = false;
    remoteSelections.clear();
    renderSource();
    window.clearTimeout(refreshTimer);
    if (event.code === 1008) {
      endAccess();
      return;
    }
    setLiveStatus(editable ? "Reconnecting…" : "Reconnecting to live project…");
    window.clearTimeout(reconnectTimer);
    reconnectTimer = window.setTimeout(connect, reconnectDelay);
    reconnectDelay = Math.min(reconnectDelay * 2, 5_000);
  });
  socket.addEventListener("error", () => socket?.close());
};

const refreshPdf = () => {
  if (!(pdf instanceof HTMLIFrameElement)) return;
  const url = new URL(pdf.src);
  url.searchParams.set("revision", String(revision));
  pdf.src = url.href;
};

const save = async () => {
  if (!editable || !editor || !savePath || saving || editor.value === savedContent) return;
  saving = true;
  pending = false;
  setSaveStatus("Saving…");
  try {
    const content = editor.value;
    const response = await fetch(savePath, {
      method: "PATCH",
      credentials: "same-origin",
      headers: { "content-type": "application/json" },
      body: JSON.stringify({ content, revision }),
    });
    if (response.status === 409) {
      setSaveStatus("Newer project changes exist · reload before editing further");
      blocked = true;
      editor.disabled = true;
      return;
    }
    if (response.status === 404) {
      endAccess();
      return;
    }
    if (!response.ok) throw new Error("Save failed");
    const snapshot = await response.json();
    if (!Number.isSafeInteger(snapshot.revision)) throw new Error("Invalid save response");
    revision = snapshot.revision;
    savedContent = content;
    root.dataset.sharedRevision = String(revision);
    remoteSelections.clear();
    renderSource();
    setSaveStatus(`Saved · revision ${revision}`);
    refreshPdf();
  } catch {
    setSaveStatus("Could not save · retrying after the next change");
  } finally {
    saving = false;
    if (!blocked && pending) scheduleSave();
  }
};

const scheduleSave = () => {
  if (!editable || blocked) return;
  pending = true;
  window.clearTimeout(saveTimer);
  saveTimer = window.setTimeout(() => void save(), 500);
};

if (editable) {
  editor?.addEventListener("input", () => {
    renderSource();
    setSaveStatus("Unsaved changes");
    scheduleSave();
  });
  for (const eventName of ["focus", "input", "keyup", "select"]) {
    editor?.addEventListener(eventName, scheduleSelectionBroadcast);
  }
  window.setInterval(async () => {
    if (!snapshotPath || saving || editor?.value !== savedContent) return;
    try {
      const response = await fetch(snapshotPath, { credentials: "same-origin" });
      if (response.status === 404) {
        endAccess();
        return;
      }
      if (!response.ok) return;
      const snapshot = await response.json();
      if (Number.isSafeInteger(snapshot.revision) && snapshot.revision > revision) location.reload();
    } catch {
      // A later poll or local save will restore status.
    }
  }, 2_000);
}

editor?.addEventListener("scroll", syncHighlightScroll);
renderSource();
connect();
